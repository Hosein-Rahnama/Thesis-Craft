% Default Path to Figures

% Page Layout
\newgeometry{top=2cm, right=2.25cm, left=2cm, bottom=2cm}

% Line Spacing of Body Text
\setstretch{1.25}

% Item Spacing
\setlist{nosep, noitemsep, leftmargin=*}

% Bullet Item Shape
\renewcommand{\labelitemi}{$\small\bullet$}

% Spacing Around Float Objects such as Tables and Figures
\setlength{\intextsep}{10pt}

% Titles for List of Tables and Figures
\AtBeginDocument{
	\renewcommand{\listtablename}{فهرست جدول‌ها}
	\renewcommand{\listfigurename}{فهرست شکل‌ها}
}

% Pesian Names for Referencing Sections, Tables and Figures
\crefname{section}{بخش}{Section}
\crefname{table}{جدول}{Section}
\crefname{figure}{شکل}{Section}

% Reset Foonote Counter per Page
\zmakeperpage{footnote}

% Commands

% Dictionary

	\centerline{\noindent\textbf{\large #1 }}

}

}
